The Assignment¶

Take a ZIP file) of images and process them, using a library built into python that you need to learn how to use. A ZIP file takes several different files and compresses them, thus saving space, into one single file. The files in the ZIP file we provide are newspaper images (like you saw in week 3). Your task is to write python code which allows one to search through the images looking for the occurrences of keywords and faces. E.g. if you search for "pizza" it will return a contact sheet of all of the faces which were located on the newspaper page which mentions "pizza". This will test your ability to learn a new (library), your ability to use OpenCV to detect faces, your ability to use tesseract to do optical character recognition, and your ability to use PIL to composite images together into contact sheets.

Each page of the newspapers is saved as a single PNG image in a file called images.zip. These newspapers are in english, and contain a variety of stories, advertisements and images. Note: This file is fairly large (~200 MB) and may take some time to work with, I would encourage you to use small_img.zip for testing.

def make_contact_list(faces_list):
    first_image = faces_list[0]
    if len(faces_list) > 5:
        img_height = 2
    else:
        img_height = 1
    contact_sheet=Image.new(first_image.mode, (first_image.width*5,first_image.height*img_height))

    x = 0
    y = 0

    for img in faces_list:
        # Lets paste the current image into the contact sheet
        contact_sheet.paste(img, (x, y) )
        # Now we update our X position. If it is going to be the width of the image, then we set it to 0
        # and update Y as well to point to the next "line" of the contact sheet.
        if x+first_image.width == contact_sheet.width:
            x=0
            y=y+first_image.height
        else:
            x=x+first_image.width
    
    return contact_sheet

def crop_faces(img, cv_img):
    faces = face_cascade.detectMultiScale(cv_img, 1.3, 5)   ## detecting faces
    faces_list = list()
    size = (100,100)
    for i in range(len(faces)):
        x,y,w,h =faces[i]
        face_img = img.crop((x, y, w+x, y+h))
        face_img.thumbnail(size)
        faces_list.append(face_img)
    
    return faces_list

def unzip_files(loc):
    fileloc = loc
    mydict = {}
    with zipfile.ZipFile(loc) as f:
        files = f.infolist()
        for file in files:
            with f.open(file) as imgfile:

                # Opening Image using PIL.Image
                img = Image.open(imgfile)
                img.save(file.filename)  # Saving image using PIL.Image
                # Opening image using CV
                cv_img = cv.imread(file.filename)    # RGB Pixel valued matrix

                mydict[file.filename] = [img, cv_img]
                

    return mydict

def extracting_lines_from_image(img):
    
    ##### Getting text from image
    gray = img.convert('1')
    return pytesseract.image_to_string(gray)


def make_dict(loc):
    mydict = unzip_files(loc)
    filenames = mydict.keys()
    for filename in filenames:
        img = mydict[filename][0]
        cv_img = mydict[filename][1]
        faces_list = crop_faces(img, cv_img) # Cropping Facing from image
        if len(faces_list) == 0:
            message = 'But there were no faces in that file!'
            mydict[filename].append(message)
        else:
            contact_sheet = make_contact_list(faces_list) # making contact sheet
            mydict[filename].append(contact_sheet)
        text = extracting_lines_from_image(img)
        mydict[filename].append(text)
        
    return mydict
    
def main(loc):
    keyword = input('Enter keyword to search for images : ')
    mydict = make_dict(loc)
    for key in mydict:
        text = mydict[key][3]
        if keyword in text:
            contact_sheet = mydict[key][2]
            print('Result found in file', key)
            if isinstance(contact_sheet, str):
                print(contact_sheet)
            else:
                display(contact_sheet)
